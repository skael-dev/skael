# Skael CLI — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `skael` CLI binary — a Go command-line tool using Cobra that communicates with the skael platform API for skill management, sync, security scanning, and activation tracking hook installation.

**Architecture:** Single Go binary using Cobra for command routing, Lipgloss for styled terminal output, and a lightweight HTTP client for API communication. Config and state stored in `~/.skael/`. Agent detection and hook installation handle Claude Code and Codex CLI. Reuses `internal/scan` for local security scanning and `internal/skill` for archive packing.

**Tech Stack:** Go 1.25, Cobra (`github.com/spf13/cobra`), Lipgloss (`github.com/charmbracelet/lipgloss`), standard `net/http` for API client.

## API Surface (from Plan 1 backend)

The CLI talks to these endpoints, all requiring `X-API-Key` header:

| Method | Path | Used by |
|---|---|---|
| GET | `/api/health` | setup (validate connection) |
| GET | `/api/skills` | list |
| GET | `/api/skills/{name}` | publish (check exists) |
| POST | `/api/skills` | publish (create skill) |
| POST | `/api/skills/{name}/versions` | publish (upload archive) |
| GET | `/api/skills/{name}/versions/{v}/download` | sync (download) |
| GET | `/api/search?q=...` | search |
| GET | `/api/sync/manifest` | sync (diff state) |
| POST | `/api/events` | hook script (activation tracking) |

```go
package config

import (
	"encoding/json"
	"fmt"
	"os"
	"path/filepath"
)

type Config struct {
	Endpoint string `json:"endpoint"`
	APIKey   string `json:"api_key"`
}

type SyncState struct {
	LastSync string        `json:"last_sync"`
	Skills   []SyncedSkill `json:"skills"`
}

type SyncedSkill struct {
	Name     string `json:"name"`
	Version  int    `json:"version"`
	Checksum string `json:"checksum"`
}

func DefaultDir() string {
	home, err := os.UserHomeDir()
	if err != nil {
		return ""
	}
	return filepath.Join(home, ".skael")
}

func WriteConfig(dir string, cfg *Config) error {
	if err := os.MkdirAll(dir, 0755); err != nil {
		return fmt.Errorf("creating config dir: %w", err)
	}
	data, err := json.MarshalIndent(cfg, "", "  ")
	if err != nil {
		return err
	}
	return os.WriteFile(filepath.Join(dir, "config.json"), data, 0600)
}

func ReadConfig(dir string) (*Config, error) {
	data, err := os.ReadFile(filepath.Join(dir, "config.json"))
	if err != nil {
		return nil, fmt.Errorf("reading config: %w", err)
	}
	var cfg Config
	if err := json.Unmarshal(data, &cfg); err != nil {
		return nil, fmt.Errorf("parsing config: %w", err)
	}
	return &cfg, nil
}

func WriteState(dir string, state *SyncState) error {
	if err := os.MkdirAll(dir, 0755); err != nil {
		return fmt.Errorf("creating config dir: %w", err)
	}
	data, err := json.MarshalIndent(state, "", "  ")
	if err != nil {
		return err
	}
	return os.WriteFile(filepath.Join(dir, "state.json"), data, 0644)
}

func ReadState(dir string) (*SyncState, error) {
	path := filepath.Join(dir, "state.json")
	data, err := os.ReadFile(path)
	if os.IsNotExist(err) {
		return &SyncState{}, nil
	}
	if err != nil {
		return nil, fmt.Errorf("reading state: %w", err)
	}

	var state SyncState
	if err := json.Unmarshal(data, &state); err != nil {
		backup := path + ".bak"
		os.Rename(path, backup)
		return &SyncState{}, nil
	}
	return &state, nil
}

func LoadConfig() (*Config, error) {
	dir := DefaultDir()
	if envURL := os.Getenv("SKAEL_URL"); envURL != "" {
		envKey := os.Getenv("SKAEL_KEY")
		if envKey != "" {
			return &Config{Endpoint: envURL, APIKey: envKey}, nil
		}
	}
	return ReadConfig(dir)
}
```

```go
package client

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"net/url"
	"strconv"
	"time"
)

type Client struct {
	endpoint   string
	apiKey     string
	httpClient *http.Client
}

func New(endpoint, apiKey string) *Client {
	return &Client{
		endpoint:   endpoint,
		apiKey:     apiKey,
		httpClient: &http.Client{Timeout: 30 * time.Second},
	}
}

type Skill struct {
	Name          string          `json:"name"`
	Description   string          `json:"description"`
	LatestVersion int             `json:"latest_version"`
	Frontmatter   json.RawMessage `json:"frontmatter,omitempty"`
	CreatedAt     time.Time       `json:"created_at"`
	UpdatedAt     time.Time       `json:"updated_at"`
}

type Version struct {
	Version    int             `json:"version"`
	Checksum   string          `json:"checksum"`
	Changelog  string          `json:"changelog"`
	ScanResult json.RawMessage `json:"scan_result,omitempty"`
	CreatedAt  time.Time       `json:"created_at"`
}

type ManifestEntry struct {
	Name     string `json:"name"`
	Version  int    `json:"version"`
	Checksum string `json:"checksum"`
}

type APIError struct {
	StatusCode int
	Message    string
}

func (e *APIError) Error() string {
	return fmt.Sprintf("API error %d: %s", e.StatusCode, e.Message)
}

func (c *Client) do(method, path string, body io.Reader, contentType string) (*http.Response, error) {
	u := c.endpoint + path
	req, err := http.NewRequest(method, u, body)
	if err != nil {
		return nil, err
	}
	req.Header.Set("X-API-Key", c.apiKey)
	if contentType != "" {
		req.Header.Set("Content-Type", contentType)
	}
	return c.httpClient.Do(req)
}

func (c *Client) Health() error {
	resp, err := c.do("GET", "/api/health", nil, "")
	if err != nil {
		return fmt.Errorf("connection failed: %w", err)
	}
	defer resp.Body.Close()
	if resp.StatusCode != 200 {
		return fmt.Errorf("health check returned %d", resp.StatusCode)
	}
	return nil
}

func (c *Client) ListSkills(limit, offset int) ([]Skill, int, error) {
	path := fmt.Sprintf("/api/skills?limit=%d&offset=%d", limit, offset)
	resp, err := c.do("GET", path, nil, "")
	if err != nil {
		return nil, 0, err
	}
	defer resp.Body.Close()

	var result struct {
		Skills []Skill `json:"skills"`
		Total  int     `json:"total"`
	}
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return nil, 0, err
	}
	return result.Skills, result.Total, nil
}

func (c *Client) GetSkill(name string) (*Skill, error) {
	resp, err := c.do("GET", "/api/skills/"+url.PathEscape(name), nil, "")
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	if resp.StatusCode == 404 {
		return nil, nil
	}
	if resp.StatusCode != 200 {
		return nil, &APIError{StatusCode: resp.StatusCode, Message: "failed to get skill"}
	}

	var skill Skill
	if err := json.NewDecoder(resp.Body).Decode(&skill); err != nil {
		return nil, err
	}
	return &skill, nil
}

func (c *Client) CreateSkill(name, description string) (*Skill, error) {
	body, _ := json.Marshal(map[string]string{"name": name, "description": description})
	resp, err := c.do("POST", "/api/skills", bytes.NewReader(body), "application/json")
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	if resp.StatusCode != 201 {
		respBody, _ := io.ReadAll(resp.Body)
		return nil, &APIError{StatusCode: resp.StatusCode, Message: string(respBody)}
	}

	var skill Skill
	if err := json.NewDecoder(resp.Body).Decode(&skill); err != nil {
		return nil, err
	}
	return &skill, nil
}

func (c *Client) PublishVersion(name string, archive []byte) (*Version, json.RawMessage, error) {
	resp, err := c.do("POST", "/api/skills/"+url.PathEscape(name)+"/versions",
		bytes.NewReader(archive), "application/gzip")
	if err != nil {
		return nil, nil, err
	}
	defer resp.Body.Close()

	respBody, _ := io.ReadAll(resp.Body)

	if resp.StatusCode == 422 {
		return nil, respBody, &APIError{StatusCode: 422, Message: "blocked by security scan"}
	}
	if resp.StatusCode != 201 {
		return nil, nil, &APIError{StatusCode: resp.StatusCode, Message: string(respBody)}
	}

	var version Version
	if err := json.Unmarshal(respBody, &version); err != nil {
		return nil, nil, err
	}
	return &version, nil, nil
}

func (c *Client) SearchSkills(query string, limit int) ([]Skill, error) {
	path := "/api/search?q=" + url.QueryEscape(query) + "&limit=" + strconv.Itoa(limit)
	resp, err := c.do("GET", path, nil, "")
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()

	var result struct {
		Skills []Skill `json:"skills"`
	}
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return nil, err
	}
	return result.Skills, nil
}

func (c *Client) GetManifest() ([]ManifestEntry, error) {
	resp, err := c.do("GET", "/api/sync/manifest", nil, "")
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()

	var entries []ManifestEntry
	if err := json.NewDecoder(resp.Body).Decode(&entries); err != nil {
		return nil, err
	}
	return entries, nil
}

func (c *Client) DownloadVersion(name string, version int) ([]byte, error) {
	path := fmt.Sprintf("/api/skills/%s/versions/%d/download", url.PathEscape(name), version)
	resp, err := c.do("GET", path, nil, "")
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	if resp.StatusCode != 200 {
		return nil, &APIError{StatusCode: resp.StatusCode, Message: "download failed"}
	}
	return io.ReadAll(resp.Body)
}
```

```go
type Agent interface {
```

```go
func formatAge(t time.Time) string {
	d := time.Since(t)

	case d < time.Hour:

	case d < 24*time.Hour:
		return fmt.Sprintf("%dh ago", int(d.Hours()))
	case d < 30*24*time.Hour:
		return fmt.Sprintf("%dd ago", int(d.Hours()/24))
	default:
		return fmt.Sprintf("%dw ago", int(d.Hours()/(24*7)))
	}
```

```go
func runSync(cmd *cobra.Command, args []string) error {
	cfg, err := config.LoadConfig()
	if err != nil {
		ui.Error(ui.ErrorDetail{Message: "Not configured", Suggestion: "skael setup <url> <api-key>"})
		return err
	}

	local, _ := config.ReadState(configDir)

	localMap := make(map[string]config.SyncedSkill)
	for _, s := range local.Skills {
		localMap[s.Name] = s
	}

	type syncAction struct {
		Name       string
		OldVersion int
		NewVersion int
		IsNew      bool
	}

	var actions []syncAction
	for _, r := range remote {
		if l, ok := localMap[r.Name]; !ok {
			actions = append(actions, syncAction{Name: r.Name, NewVersion: r.Version, IsNew: true})
		} else if r.Version > l.Version || r.Checksum != l.Checksum {
			actions = append(actions, syncAction{Name: r.Name, OldVersion: l.Version, NewVersion: r.Version})
		}
	}

	if len(actions) == 0 {
		ui.Success("Already up to date")
		if ui.JSONMode {
			ui.PrintJSON(map[string]interface{}{"synced": []string{}, "total": len(remote)})
		} else {
			ui.Summary(fmt.Sprintf("%d skills", len(remote)))
		}
		return nil
	}

	if syncDryRun {
		for _, a := range actions {
			if a.IsNew {
				ui.New(a.Name, fmt.Sprintf("v%d", a.NewVersion))
			} else {
				ui.Download(a.Name, fmt.Sprintf("v%d → v%d", a.OldVersion, a.NewVersion))
			}
		}
		ui.Summary(fmt.Sprintf("%d to sync (dry run)", len(actions)))
		return nil

	detected := agents.DetectIn(home)

	type syncResult struct {
		Name        string `json:"name"`
		FromVersion *int   `json:"from_version"`
		ToVersion   int    `json:"to_version"`

		}

			v := a.OldVersion
			r.FromVersion = &v
		}
		results = append(results, r)
	}

		LastSync: fmt.Sprintf("%s", time.Now().UTC().Format(time.RFC3339)),
```

```go
func InstallClaudeHook(configPath, endpoint, apiKey, scriptPath string) error {
	var settings map[string]interface{}

	data, err := os.ReadFile(configPath)
	if err == nil {
		json.Unmarshal(data, &settings)
	}
	if settings == nil {
		settings = map[string]interface{}{}
	}

	hooks, _ := settings["hooks"].(map[string]interface{})
	if hooks == nil {
		hooks = map[string]interface{}{}
	}

	command := fmt.Sprintf("SKAEL_AGENT=claude-code SKAEL_ENDPOINT=%s SKAEL_API_KEY=%s %s",
		endpoint, apiKey, scriptPath)

	newHookEntry := map[string]interface{}{
		"type":        "command",
		"command":     command,
		"_managed_by": "skael",
	}

	preToolUse, _ := hooks["PreToolUse"].([]interface{})

	replaced := false
	for i, entry := range preToolUse {
		e, ok := entry.(map[string]interface{})
		if !ok {
			continue
		}
		hooksList, _ := e["hooks"].([]interface{})
		for j, h := range hooksList {
			hm, ok := h.(map[string]interface{})
			if !ok {
				continue
			}
			if hm["_managed_by"] == "skael" {
				hooksList[j] = newHookEntry
				replaced = true
			}
		}
		e["hooks"] = hooksList
		preToolUse[i] = e
	}

	if !replaced {
		preToolUse = append(preToolUse, map[string]interface{}{
			"matcher": "Skill",
			"hooks":   []interface{}{newHookEntry},
		})
	}

	hooks["PreToolUse"] = preToolUse
	settings["hooks"] = hooks

	if err := os.MkdirAll(filepath.Dir(configPath), 0755); err != nil {
		return err
	}

	out, err := json.MarshalIndent(settings, "", "  ")
	if err != nil {
		return err
	}
	return os.WriteFile(configPath, out, 0644)
}

func UninstallClaudeHook(configPath string) error {
	data, err := os.ReadFile(configPath)
	if os.IsNotExist(err) {
		return nil
	}
	if err != nil {
		return err
	}

	var settings map[string]interface{}
	if err := json.Unmarshal(data, &settings); err != nil {
		return err
	}

	hooks, _ := settings["hooks"].(map[string]interface{})
	if hooks == nil {
		return nil
	}

	preToolUse, _ := hooks["PreToolUse"].([]interface{})
	var filtered []interface{}
	for _, entry := range preToolUse {
		e, ok := entry.(map[string]interface{})
		if !ok {
			filtered = append(filtered, entry)
			continue
		}
		hooksList, _ := e["hooks"].([]interface{})
		var filteredHooks []interface{}
		for _, h := range hooksList {
			hm, ok := h.(map[string]interface{})
			if !ok || hm["_managed_by"] != "skael" {
				filteredHooks = append(filteredHooks, h)
			}
		}
		if len(filteredHooks) > 0 {
			e["hooks"] = filteredHooks
			filtered = append(filtered, e)
		}
	}

	if len(filtered) > 0 {
		hooks["PreToolUse"] = filtered
	} else {
		delete(hooks, "PreToolUse")
	}

	if len(hooks) == 0 {
		delete(settings, "hooks")
	} else {
		settings["hooks"] = hooks
	}

	out, _ := json.MarshalIndent(settings, "", "  ")
	return os.WriteFile(configPath, out, 0644)
}

func InstallForAgent(agentName, configPath, endpoint, apiKey, scriptPath string) error {
	switch agentName {
	case "claude-code":
		return InstallClaudeHook(configPath, endpoint, apiKey, scriptPath)
	case "codex":
		return installCodexHook(configPath, endpoint, apiKey, scriptPath)
	default:
		return fmt.Errorf("unsupported agent: %s", agentName)
	}
}

func installCodexHook(configPath, endpoint, apiKey, scriptPath string) error {
	command := fmt.Sprintf("SKAEL_AGENT=codex SKAEL_ENDPOINT=%s SKAEL_API_KEY=%s %s",
		endpoint, apiKey, scriptPath)

	block := fmt.Sprintf(`
# managed_by = "skael"
[[hooks.pre_tool_use]]
matcher = "Skill"
command = "%s"
`, command)

	data, err := os.ReadFile(configPath)
	if os.IsNotExist(err) {
		data = []byte{}
	} else if err != nil {
		return err
	}

	content := string(data)
	if indexOf(content, `# managed_by = "skael"`) >= 0 {
		start := indexOf(content, `# managed_by = "skael"`)
		end := indexOfFrom(content, "[[", start+1)
		if end < 0 {
			end = len(content)
		}
		content = content[:start] + block[1:] + content[end:]
	} else {
		content += block
	}

	if err := os.MkdirAll(filepath.Dir(configPath), 0755); err != nil {
		return err
	}
	return os.WriteFile(configPath, []byte(content), 0644)
}

func indexOf(s, sub string) int {
	for i := 0; i <= len(s)-len(sub); i++ {
		if s[i:i+len(sub)] == sub {
			return i
		}
	}
	return -1
}

func indexOfFrom(s, sub string, from int) int {
	for i := from; i <= len(s)-len(sub); i++ {
		if s[i:i+len(sub)] == sub {
			return i
		}
	}
	return -1
}

func UninstallForAgent(agentName, configPath string) error {
	switch agentName {
	case "claude-code":
		return UninstallClaudeHook(configPath)
	case "codex":
		return uninstallCodexHook(configPath)
	default:
		return nil
	}
}
```

```go
func runSetup(cmd *cobra.Command, args []string) error {
	var endpoint, apiKey string

	if len(args) >= 2 {
		endpoint = args[0]
		apiKey = args[1]
	} else {
		endpoint = os.Getenv("SKAEL_URL")
		apiKey = os.Getenv("SKAEL_KEY")
	}

	if endpoint == "" || apiKey == "" {
		ui.Error(ui.ErrorDetail{
			Message: "URL and API key required",
			Context: "Provide as arguments or set SKAEL_URL and SKAEL_KEY environment variables.",
			Suggestion: "skael setup https://skills.company.com sk-xxxxxxxxxxxxx",
		})
		return fmt.Errorf("missing url or api key")
	}

	c := client.New(endpoint, apiKey)
	if err := c.Health(); err != nil {
		ui.Error(ui.ErrorDetail{
			Message:    fmt.Sprintf("Cannot connect to %s", endpoint),
			Context:    err.Error(),
			Suggestion: fmt.Sprintf("curl -s %s/api/health", endpoint),
		})
		return err
	}
	ui.Success("Connected to %s", endpoint)

	configDir := config.DefaultDir()
	cfg := &config.Config{Endpoint: endpoint, APIKey: apiKey}
	if err := config.WriteConfig(configDir, cfg); err != nil {
		ui.Error(ui.ErrorDetail{Message: "Failed to write config", Context: err.Error()})
		return err
	}
	ui.Success("Config written to %s", configDir)

	home, _ := os.UserHomeDir()
	detected := agents.DetectIn(home)
	if len(detected) > 0 {
		names := make([]string, len(detected))
		for i, a := range detected {
			names[i] = a.Name()
		}
		ui.Success("Detected agents: %s", joinNames(names))
	} else {
		ui.Warn("No supported agents detected")
	}

	if !setupSkipSync {
		if err := runSync(cmd, nil); err != nil {
			ui.Warn("Sync failed: %v (you can retry with skael sync)", err)
		}
	}

	if !setupSkipHooks && len(detected) > 0 {
		scriptPath, err := hooks.WriteHookScript(configDir)
		if err != nil {
			ui.Warn("Failed to write hook script: %v", err)
		} else {
			for _, agent := range detected {
				err := hooks.InstallForAgent(agent.Name(), agent.ConfigPath(home), endpoint, apiKey, scriptPath)
				if err != nil {
					ui.Warn("Hook install failed for %s: %v", agent.Name(), err)
				} else {
					ui.Success("Installed hook for %s", agent.Name())
				}
			}
		}
	}

	if !ui.JSONMode {
		fmt.Println()
		fmt.Println("  Setup complete. Skills are live.")
		fmt.Println("  Run " + ui.Code("skael sync") + " anytime to update.")
	}
	return nil
}
```
